 522        # Update the stream stats table.
 523        last_row = dict(await bot.db.get_last_row('stream_stats'))
 524        await bot.db.update(
 525            'stream_stats',
 526            {'rewards': last_row['rewards'] + 1},
 527            {'id': last_row['id']},
 528        )
 529
 530        parts = self.reward_title.split(' ')
 531        command = parts[0].lower()
 532        command_args = []
 533
 534        # Some arguments might be included in the reward name.

 536            command_args.extend(parts[1:])
 537
 538        # Rewards can offer a prompt to the user, so if a reply exists,
 539        # add it to the arguments.
 540        if self.reward_prompt:
 541            command_args.append(self.reward_prompt)
 542
 543        # If a plugin is associated with the reward, run it.
 544        redemption_plugins = bot.plugins.get('redemption', {})
 545        if command in redemption_plugins:
 546            plugin = redemption_plugins[command]
 547            user = {'name': self.user_name, 'id': self.user_id}
 548
 549            LOG.debug(f'Running {plugin} for {self.reward_title} redemption')
 550            await bot._run_plugin(plugin, self, user, command, command_args)
