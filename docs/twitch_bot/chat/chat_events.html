 34class ChatEvent(Event):
 35    """Message event.
 36
 37    This object breakes down a server message into its parts and makes it
 38    easier to access the individual elements.
 39
 40    Do not use this class directly, only subclass from it.
 41    """
 42
 43    def __init__(self) -> None:
 44        """Init."""
 45        super(ChatEvent, self).__init__()
 46
 47    async def init(self, data: str):
 48        """Async init.
 49
 50        Args:
 51            data (str): Data from the server.
 52        """
 53        self.data = data
 54        if not self.data:
 55            LOG.warning('Event has no data.')
 56            return
 57
 58        # Attempt to break down the message into parts.
 59        matches = re.match(MESSAGE_RE, self.data)
 60        match_dict = matches.groupdict() if matches else {}
 61
 62        # Metadata from the server data.
 63        metadata = match_dict.get('metadata')
 64        data_dict = {}
 65        if metadata:
 66            for datum in metadata.split(';'):
 67                key, value = datum.split('=')
 68                data_dict[key] = value
 69
 70        self.badge_info = data_dict.get('badge-info', '')
 71        self.badges = data_dict.get('badges', '')
 72        self.client_nonce = data_dict.get('client-nonce', '')
 73        self.color = data_dict.get('color', '')
 74        self.display_name = data_dict.get('display-name', '')
 75        self.emotes = data_dict.get('emotes', '')
 76        self.flags = data_dict.get('flags', '')
 77        self.id = data_dict.get('id', '')
 78        self.mod = False if int(data_dict.get('mod', 0)) == 0 else True
 79        self.room_id = int(data_dict.get('room-id', 0))
 80        self.subscriber = False if int(data_dict.get('mod', 0)) == 0 else True
 81        self.tmi_sent_ts = int(data_dict.get('tmi-sent-ts', 0))
 82        self.turbo = False if int(data_dict.get('mod', 0)) == 0 else True
 83        self.user_id = int(data_dict.get('user-id', 0))
 84        self.user_type = data_dict.get('user-type', '')
 85
 86        # Event data from the server message.
 87        self.username = match_dict.get('username', '')
 88        self.address = match_dict.get('address', '')
 89        self.type = match_dict.get('type', '')
 90        self.channel = match_dict.get('channel', '')
 91        self.message = match_dict.get('message', '')
 92
 93        return

142        """Run the code triggered by the event.
143
144        Args:
145            bot (Bot): Bot instance.
146        """
147        LOG.debug(f'A name list was received: {self}.')
148        raw_username_list = self.message.split(':')
149        if not raw_username_list or len(raw_username_list) < 2:
150            return

152        username_list = [

154            for u in raw_username_list[1].split()
155            if u not in [bot.owner, bot.username]

158        for username in username_list:

161            LOG.debug(f'user_data: {user_data}')
162            if not user_data:
163                return
164
165            # Get the user data from the database.
166            user_db_data = await bot._get_user_db_data(user_data)
167
168            # Otherwise, update the user data in the database.
169            update_dict = {'last_join_time': str(datetime.now())}
170            await bot.db.update(
171                'users', update_dict, {'username': user_db_data['username']}
172            )
173
174            # Add user to cache.
175            user_cache = await CACHE.get('users')
176            if username not in user_cache.data:
177                user_cache.data.append(username)

363        # Get the username from the event.
364        username = self.username
365        LOG.info(f'{username} left.')
366
367        # Get the user data from the username.

383
384        update_dict = {
385            'time_watched': time_watched,
386            'last_leave_time': last_leave_time,
387            'messages_sent_session': 0,
388            'messages_sent_total': (
389                int(user_db_data['messages_sent_total'])
390                + int(user_db_data['messages_sent_session'])
391            ),
392        }
393
394        # Update the database.
395        await bot.db.update(
396            'users', update_dict, {'user_id': user_db_data['user_id']}
397        )
398
399        # Remove user from cache.
400        user_cache = await CACHE.get('users')
401        if username in user_cache.data:
402            user_cache.data.remove(username)

405class PubmsgEvent(ChatEvent):
406    """Receive a message event.
407
408    This is called any time a message is published in the chat.
409    """
410
411    def __init__(self) -> None:
412        """Init."""
413        super(PubmsgEvent, self).__init__()
414
415    async def init(self, data: str) -> object:
416        """Async init.
417
418        Args:
419            data (str): Data from the server.
420        """
421        await super(PubmsgEvent, self).init(data)
422        return self
423
424    async def run(self, bot: object) -> None:
425        """Run the code triggered by the event.
426
427        Args:
428            bot (Bot): Bot instance.
429        """
430        # Parse the event.
431        user = {'name': self.username, 'id': self.user_id}
432        message = self.message
433        LOG.info(f'Message from {user["name"]}: {message}')
434
435        bot.last_message_time = datetime.now()
436
437        # Get the user data from the username.
438        user_data = await api.get_user_data(bot.auth, user['name'])
439        LOG.debug(f'user_data: {user_data}')
440        if not user_data:
441            return
442
443        # Get the user data from the database.
444        user_db_data = await bot._get_user_db_data(user_data)
445
446        # Run the message past the Moderator.
447        mod_approved = await bot.moderator.check_message(user, message)
448        if not mod_approved:
449            LOG.debug('Message rejected.')
450            return
451
452        # If a chat message starts with an exclamation point,
453        # try to run it as a command.
454        if message.startswith('!'):
455            split_message = message.split(' ', maxsplit=1)
456            command = split_message[0].lstrip('!')
457
458            if len(split_message) >= 2:
459                command_args = split_message[1].split(' ')
460            else:
461                command_args = None
462
463            await bot.dispatch_command(self, user, command, command_args)
464
465        # If the message doesn't start with a command, send the message to the
466        # text parser.
467        else:
468            await bot.dispatch_message(self, user, message)
469
470        # Welcome first time chatters.
471        if user_db_data['messages_sent_total'] == 0:
472            LOG.debug('First time chatter.')
473            await bot.send_message(f'Welcome @{self.username}!')
474
475        # Increment the number of sent messages in the database.
476        users_update_dict = {
477            'messages_sent_session': user_db_data['messages_sent_session'] + 1
478        }
479
480        await bot.db.update(
481            'users', users_update_dict, {'user_id': user_db_data['user_id']}
482        )
483
484        # Update the stream stats database.
485        stream_stats_last_row = dict(await bot.db.get_last_row('stream_stats'))
486
487        await bot.db.update(
488            'stream_stats',
489            {'messages': int(stream_stats_last_row.get('messages', 0)) + 1},
490            {'id': stream_stats_last_row.get('id', 0)},
491        )
492
493        # Add unique chatters to the cache.
494        chatters_cache = await CACHE.get('chatters')
495        if self.username not in chatters_cache.data:
496            chatters_cache.data.append(self.username)

426        """Run the code triggered by the event.
427
428        Args:
429            bot (Bot): Bot instance.
430        """
431        # Parse the event.
432        user = {'name': self.username, 'id': self.user_id}
433        message = self.message
434        LOG.info(f'Message from {user["name"]}: {message}')
435
436        bot.last_message_time = datetime.now()
437
438        # Get the user data from the username.
439        user_data = await api.get_user_data(bot.auth, user['name'])
440        LOG.debug(f'user_data: {user_data}')
441        if not user_data:
442            return
443
444        # Get the user data from the database.
445        user_db_data = await bot._get_user_db_data(user_data)
446
447        # Run the message past the Moderator.
448        mod_approved = await bot.moderator.check_message(user, message)
449        if not mod_approved:
450            LOG.debug('Message rejected.')
451            return
452
453        # If a chat message starts with an exclamation point,
454        # try to run it as a command.
455        if message.startswith('!'):
456            split_message = message.split(' ', maxsplit=1)
457            command = split_message[0].lstrip('!')
458
459            if len(split_message) >= 2:
460                command_args = split_message[1].split(' ')
461            else:
462                command_args = None
463
464            await bot.dispatch_command(self, user, command, command_args)
465
466        # If the message doesn't start with a command, send the message to the
467        # text parser.
468        else:
469            await bot.dispatch_message(self, user, message)
470
471        # Welcome first time chatters.
472        if user_db_data['messages_sent_total'] == 0:
473            LOG.debug('First time chatter.')
474            await bot.send_message(f'Welcome @{self.username}!')
475
476        # Increment the number of sent messages in the database.
477        users_update_dict = {
478            'messages_sent_session': user_db_data['messages_sent_session'] + 1
479        }
480
481        await bot.db.update(
482            'users', users_update_dict, {'user_id': user_db_data['user_id']}
483        )
484
485        # Update the stream stats database.
486        stream_stats_last_row = dict(await bot.db.get_last_row('stream_stats'))
487
488        await bot.db.update(
489            'stream_stats',
490            {'messages': int(stream_stats_last_row.get('messages', 0)) + 1},
491            {'id': stream_stats_last_row.get('id', 0)},
492        )
493
494        # Add unique chatters to the cache.
495        chatters_cache = await CACHE.get('chatters')
496        if self.username not in chatters_cache.data:
497            chatters_cache.data.append(self.username)

439        user_data = await api.get_user_data(bot.auth, user['name'])
440        LOG.debug(f'user_data: {user_data}')
441        if not user_data:
442            return
443
444        # Get the user data from the database.
445        user_db_data = await bot._get_user_db_data(user_data)
446
447        # Run the message past the Moderator.
448        mod_approved = await bot.moderator.check_message(user, message)
449        if not mod_approved:
450            LOG.debug('Message rejected.')
451            return
452
453        # If a chat message starts with an exclamation point,
454        # try to run it as a command.
455        if message.startswith('!'):
456            split_message = message.split(' ', maxsplit=1)
457            command = split_message[0].lstrip('!')
458
459            if len(split_message) >= 2:
460                command_args = split_message[1].split(' ')
461            else:
462                command_args = None
463
464            await bot.dispatch_command(self, user, command, command_args)
465
466        # If the message doesn't start with a command, send the message to the
467        # text parser.
468        else:
469            await bot.dispatch_message(self, user, message)
470
471        # Welcome first time chatters.
472        if user_db_data['messages_sent_total'] == 0:
473            LOG.debug('First time chatter.')
474            await bot.send_message(f'Welcome @{self.username}!')
475
476        # Increment the number of sent messages in the database.
477        users_update_dict = {
478            'messages_sent_session': user_db_data['messages_sent_session'] + 1
479        }
480
481        await bot.db.update(
482            'users', users_update_dict, {'user_id': user_db_data['user_id']}
483        )
484
485        # Update the stream stats database.
486        stream_stats_last_row = dict(await bot.db.get_last_row('stream_stats'))
487
488        await bot.db.update(
489            'stream_stats',
490            {'messages': int(stream_stats_last_row.get('messages', 0)) + 1},
491            {'id': stream_stats_last_row.get('id', 0)},
492        )
493
494        # Add unique chatters to the cache.
495        chatters_cache = await CACHE.get('chatters')
496        if self.username not in chatters_cache.data:
497            chatters_cache.data.append(self.username)
